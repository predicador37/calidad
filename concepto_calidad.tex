	\begin{frame}
		\frametitle{Definición de calidad}
	La calidad se ha convertido en uno de los principales objetivos estratégicos para las organizaciones, dado que su supervivencia cada vez depende más de la calidad de sus productos y servicios. Según el DRAE, la calidad es:
	\begin{enumerate}
		\item \textit{f. Propiedad o conjunto de propiedades inherentes a algo, que permiten juzgar su valor. Esta tela es de buena calidad}.
		\item\textit{ f. Buena calidad, superioridad o excelencia. La calidad de ese aceite ha conquistado los mercados.}
		\item \textit{f. Adecuación de un producto o servicio a las características especificadas. Control de la calidad de un producto.}
		\item\textit{ f. Carácter, genio, índole.}
	    \item \textit{f. Condición o requisito que se pone en un contrato.}
	\end{enumerate}
	
		
	\end{frame}

	\begin{frame}
	\frametitle{Vistas de la calidad (Garbin, 1984)}
\begin{itemize}
 \item \textbf{Vista trascendental}: la calidad es algo que se reconoce pero no se define: un ideal al que se intenta llegar.
 \item \textbf{Vista del usuario}: la calidad es adecuación al propósito. Las características de los productos se pueden cuantivicar.
 \item \textbf{Vista del fabricante}: la calidad es conformidad con las especificaciones. Vista centrada en el proceso.
 \item \textbf{Vista del producto}: la calidad está unida a las características inherentes del producto.
 \item \textbf{Vista basada en valor}: la calidad depende de la cantidad que el cliente esté dispuesto a pagar.
\end{itemize}
	
\end{frame}

\section{Conceptos relacionados con la calidad}

	\begin{frame}
	\frametitle{Conceptos relacionados con gestión de calidad (ISO 9000)}
	\begin{itemize}
		\item \textbf{Política de la calidad}: intenciones y dirección de una organización relativas a la calidad como las expresa su alta dirección.
		\item \textbf{Objetivo de la calidad}: resultado a lograr en cuanto a calidad.
		\item \textbf{Sistema de gestión de la calidad}: sistema para dirigir y controlar una organización con respecto a la calidad.
		\item \textbf{Planificación de la calidad}: establecimiento de objetivos y especificación de procesos y recursos necesarios para cumplir con los objetivos de la calidad.
		\item \textbf{Control de la calidad}: orientada al cumplimiento de requisitos de calidad.
		\item \textbf{Aseguramiento de la calidad}: orientada a proporcionar confianza en que se cumplirán los requisitos de la calidad.
		\item \textbf{Mejora de la calidad}: orientada a aumentar la capacidad para cumplir con los requisitos de la calidad.
	\end{itemize}
	
\end{frame}

	\begin{frame}
	\frametitle{Conceptos relacionados con aceptación de producto o servicio (ISO 9000)}
	\begin{itemize}
		\item \textbf{Conformidad}: cumplimiento de un requisito.
		\item \textbf{No conformidad}: incumplimiento de un requisito.
		\item \textbf{Defecto}: no conformidad relativa a un uso previsto o especificado.
		\item \textbf{Acción preventiva}: tomada para eliminar la causa de una no conformidad potencial.
		\item \textbf{Acción correctiva}: tomada para eliminar la causa de una no conformidad y prevenir su ocurrencia.
		\item \textbf{Corrección}: acción tomada para eliminar una no conformidad detectada.
		\item \textbf{Reparación}: acción tomada sobre un producto no conforme para convertirlo en aceptable para su utilización prevista.
		\item \textbf{Desecho}: acción tomada sobre un producto no conforme para impedir su uso inicialmente previsto.
	\end{itemize}
	
\end{frame}

	\begin{frame}
	\frametitle{Conceptos relacionados con la documentación de la calidad}
	\begin{itemize}
		\item \textbf{Manuales de la calidad}: proporcionan información coherente sobre el sistema de gestión de la calidad de la organización.
		\item \textbf{Planes de la calidad}: describen cómo se aplica el sistema de gestión de la calidad a un producto o proyecto.
		\item \textbf{Especificaciones}: establecen requisitos.
		\item \textbf{Guías}: establecen recomendaciones o sugerencias.
		\item \textbf{Procedimientos documentados}: instrucciones de trabajo y documentos que proporcionan información sobre cómo efectuar las actividades y procesos de forma coherente.
		\item \textbf{Registros}: documentos que proporcionan evidencia objetiva de las actividades realizadas o resultados obtenidos.
	
	\end{itemize}
	
\end{frame}
